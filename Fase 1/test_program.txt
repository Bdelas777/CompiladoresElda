program Funciones;
var
  resultado : int;

void suma(a : int, b : int) {
  var
    result : int;
  result = a + b;
  print("La suma es: ", result);
}

void factorial(n : int) {
  var
    i, fact : int;
  i = 1;
  fact = 1;
  
  while (i <= n) do {
    fact = fact * i;
    i = i + 1;
  };
  
  print("El factorial de ", n, " es: ", fact);
}

main
{
  suma(5, 7);
  factorial(5);
}
end
